<script>
function showQuestions(){
    document.getElementById("questions").style.display = "block";
}

function calculate(){
    let checkboxes = document.querySelectorAll("#questions input[type='checkbox']:checked");
    let score = 0;
    checkboxes.forEach(c => score += parseInt(c.value));
    document.getElementById("scoreInput").value = score;

    let resultDiv = document.getElementById("result");
    document.getElementById("insertBtn").style.display = "block";

    if(score >= 60){
        resultDiv.className = "result good";
        resultDiv.innerHTML = "✔ نعم، هذه القطعة مناسبة للشراء 👌";
    } else {
        resultDiv.className = "result bad";
        resultDiv.innerHTML = "✖ لا، من الأفضل عدم شراء هذه القطعة 😕";
    }
}
